WITH cte AS (
    SELECT COUNT(DISTINCT order_id) AS total_orders
    FROM olist_orders_dataset
    WHERE order_purchase_timestamp BETWEEN '2018-01-01' AND '2018-01-31'
),
cte2 AS (
    SELECT COUNT(DISTINCT ood.order_id) AS count_orders
    FROM olist_orders_dataset AS ood
    JOIN olist_order_reviews_dataset AS ord ON ood.order_id = ord.order_id
    WHERE ood.order_purchase_timestamp BETWEEN '2018-01-01' AND '2018-01-31'
      AND ord.review_score = 5
)
SELECT
    count_orders,
    total_orders,
    ROUND(100.0 * count_orders / total_orders, 2) AS percentage
FROM cte, cte2;
''')

c.execute('''WITH first_orders AS (
    SELECT
        c.customer_unique_id,
        o.order_id,
        o.order_purchase_timestamp,
        ROW_NUMBER() OVER (
            PARTITION BY c.customer_unique_id
            ORDER BY o.order_purchase_timestamp
        ) as rn
    FROM olist_customers_dataset c
    JOIN olist_orders_dataset o
        ON c.customer_id = o.customer_id
),

first_order_items AS (
    SELECT
        f.customer_unique_id,
        p.product_category_name
    FROM first_orders f
    JOIN olist_order_items_dataset i
        ON f.order_id = i.order_id
    JOIN olist_products_dataset p
        ON i.product_id = p.product_id
    WHERE f.rn = 1
)

SELECT
    product_category_name,
    COUNT(DISTINCT customer_unique_id) as customer_count
FROM first_order_items
GROUP BY product_category_name
ORDER BY customer_count DESC;
''')

Hypothesis 2: Boosting 2nd-Order AOV

This test determines if offering free shipping for orders over $150 specifically targets and increases the spending of customers returning for their second purchase.

Business Goal: Increase the Average Order Value (AOV) of returning customers.

Null Hypothesis (H0): The free shipping threshold has no effect on the Average Order Value of second-time buyers.

Alternative Hypothesis (Ha): The free shipping threshold significantly increases the Average Order Value.

Target Segment: Customers who have made exactly one purchase and are making their second.

Statistical Test: Mann-Whitney U Test (chosen because e-commerce spend data is typically non-normal and right-skewed).

import numpy as np
from scipy import stats
import seaborn as sns
import matplotlib.pyplot as plt


np.random.seed(42)
control_aov = np.random.normal(loc=148.51, scale=30, size=1000)

treatment_aov = np.random.normal(loc=163.36, scale=35, size=1000)


_, p_norm_a = stats.shapiro(control_aov)
_, p_norm_b = stats.shapiro(treatment_aov)

print(f"Normality P-values: Group A={p_norm_a:.4f}, Group B={p_norm_b:.4f}")


if p_norm_a < 0.05 or p_norm_b < 0.05:
    print("Data is non-normal. Running Mann-Whitney U Test...")
    u_stat, p_val = stats.mannwhitneyu(control_aov, treatment_aov, alternative='two-sided')
else:
    print("Data is normal. Running Welch's T-Test...")
    u_stat, p_val = stats.ttest_ind(control_aov, treatment_aov, equal_var=False)

print(f"Final Test P-Value: {p_val:.10f}")


if p_val < 0.05:
    print("\nConclusion: Reject the Null Hypothesis. The threshold significantly increased AOV.")
else:
    print("\nConclusion: Fail to reject the Null Hypothesis. No significant difference found.")
